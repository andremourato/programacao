    public static void bubbleSort(int a[]) {
        int  t = 0;
        do {
            t = 0;
            for(int i = 0 ; i <= a.length - 2; i++) {
                if(a[i] > a[i+1]) {
                    swap(a, i, i+1);
                    t++;
                }
            }
        } while(t != 0);
    }
